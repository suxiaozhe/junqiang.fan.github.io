获取基金一览

import requests
import pandas as pd

if name == 'main':
url = 'http://fund.eastmoney.com/js/fundcode_search.js'
headers = {
'User-Agent': 'Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.1 (KHTML, like Gecko) Chrome/21.0.1180.71 Safari/537.1 LBBROWSER'
}
resp = requests.get(url, headers=headers).text
str = resp[8:]
list = eval(str.split(";")[0])
len_ = len(list)
latest_ = list[len_-1]

code = []
short_pinyin = []
name = []
type = []
complete_pinyin = []

for i in range(len(list)):
    code.append(list[i][0])
    short_pinyin.append(list[i][1])
    name.append(list[i][2])
    type.append(list[i][3])
    complete_pinyin.append(list[i][4])

df = pd.DataFrame()
df['基金代码'] = code
df['拼音简称'] = short_pinyin
df['名称'] = name
df['类型'] = type
df['拼音全称'] = complete_pinyin

try:
    df.to_excel(f'基金一览.xlsx', '基金一览', index=None, encoding='utf-8')

except Exception as e:
    print(e)
